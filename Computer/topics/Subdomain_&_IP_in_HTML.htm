Subdomain & IP in HTML
#!/bin/bash
 
# www.churchofsatan.com
# Arg 1 ($1) = chruchofsatan
# Arg 2 ($2) = com
 
# Download the index.html page
wget www.$1.$2
 
# Carves out URL-referenced ".$1.$2" subdomains
grep -o "[^/]*\.$1\.$2" index.html | sort -u > domain_list.txt
 
# Look for IP addresses of each domain name using the host command & extract IP addresses only
for url in $(cat domain_list.txt); do host $url; done | grep "has address" | cut -d " " -f 4 | sort -u > IP_list.txt
